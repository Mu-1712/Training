--------------------------------------------------------------------------		Using replace function with table columns
--------------------------------------------------------------------------
use db1;
go
--------------------------------------------------------------------------
CREATE TABLE cities 	(  
    				id 	INT 		PRIMARY KEY,  
    				city 	VARCHAR (55) 	NOT NULL  
			);  
go
--------------------------------------------------------------------------

Suppose we want to replace all occurrences of the character 'a' with 'A' in the 'city' column. 
--------------------------------------------------------------------------
UPDATE cities 
SET city = REPLACE(city,'e','*')  
WHERE city IS NOT NULL; 
go
--------------------------------------------------------------------------
id	city
--------------------------------------------------------------------------
1	d*lhi
2	pun*
3	hyd*rabad
4	banglor*
5	ch*nnai
6	Mumbai
--------------------------------------------------------------------------
SELECT *, REPLACE (city,'*','e') AS "Original city name" FROM cities;
go
--------------------------------------------------------------------------
id	city		Original city name
--------------------------------------------------------------------------

2	pun*		pune
3	hyd*rabad	hyderabad
